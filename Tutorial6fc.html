# Simulating data according to a normal distribution
x <- 1:100
y <- rnorm(100)
data <- data.frame(x = x, y = y)
p1 <- ggplot(data, aes(x = x, y = y)) + 
  geom_line()
p2 <- ggplot(data, aes(x = x, y = y)) + 
  geom_point()
grid.arrange(p1, p2, nrow = 1)
